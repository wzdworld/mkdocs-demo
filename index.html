随便一段python代码

import tkinter as tk


def on_button_click():
    label.config(text="Hello, " + entry.get())

def on_button_click():
    label.config()


# 创建主窗口
root = tk.Tk()
root.title("Simple GUI")

# 创建标签
label = tk.Label(root, text="Enter your name:")
label.pack()

# 创建输入框
entry = tk.Entry(root)
entry.pack()

# 创建按钮
button = tk.Button(root, text="Say Hello", command=on_button_click)
button.pack()

# 运行主循环
root.mainloop()
